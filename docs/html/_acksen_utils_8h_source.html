59beginDebug() do { DebugSerial.begin(DEBUG_BAUD_RATE); } while (0) // Serial port agnostic version of Serial.begin()
60
61// Serial port agnostic version of Serial.print()
62Trace(x) DebugSerial.print(x)
63
64// Serial port agnostic version of Serial.println()
65Traceln(x) DebugSerial.println(x)
66
67// Serial port agnostic version of Serial.print(), with inclusion of name of function it is being called in
68TraceFunc() do { DebugSerial.print ("In function: "); DebugSerial.println (__PRETTY_FUNCTION__); } while (0)

70// Serial port agnostic version of Serial.flush()
71TraceFlush() DebugSerial.flush()
72
73@endcode
